Body mass index calculation

function bmi(w, h) {
 let indexHeight = Math.pow(h, 2);
 let index = w / indexHeight;
 if(index <= 18.5) {
  return "Underweight";
 } else if(index <= 25.0) {
  return "Normal";
 } else if(index <= 30.0) {
  return "Overweight";
 } else {
  return "Obese";
 }
}
